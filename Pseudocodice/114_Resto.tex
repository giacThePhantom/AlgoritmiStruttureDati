\caption{\protect\Int[] \protect\Resto{\protect\Int[] t, \protect\Int n, \protect\Int R}}
\Int[] DP = \New \Int$[0\dots R]$\;
\Int[] S = \New \Int$[0\dots R]$\;
DP[0] = 0\;
\For{\Int i = 1 \To R}{
	DP[i] = +$\infty$\;
	\For{\Int j = 1 \To n}{
		\If{i $>$ t[j] \And DP[i - t[j]] + 1 $<$ DP[i]}{
			DP[i] = DP[i - t[j]] + 1\;
			S[i] = j\;
		}
	}
}
\Int[] x = \New \Int$[1\dots n]$\;
\For{\Int i = 1 \To n}{
	x[i] = 0\;
}
\While{R $>$ 0}{
	x[S[r]] = x[S[r]] + 1\;
	R -= t[S[R]]\;
}
\Return x\;
